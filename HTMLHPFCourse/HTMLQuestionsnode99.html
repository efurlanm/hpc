Local Index Sets

  1.      REAL A(10), B(10,10), C(25)
!HPF$ PROCESSORS P(7)
!HPF$ TEMPLATE T(51)
!HPF$ ALIGN A(:) WITH T(1::5)
!HPF$ ALIGN B(:,*) WITH T(20::3)
!HPF$ ALIGN C(:) WITH T(:)
!HPF$ ALIGN C(:) WITH T(:)
!HPF$ DISTRIBUTE T(CYCLIC(2)) ONTO P
