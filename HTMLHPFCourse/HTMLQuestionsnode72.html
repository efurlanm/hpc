Solution

    FUNCTION Triangular_Nos(n)
     IMPLICIT NONE
     INTEGER, INTENT(IN)   :: n
     INTEGER, DIMENSION(n) :: Triangular_Nos
     INTEGER               :: i
      Triangular_Nos(1) = 1
      DO i = 2, n
       Triangular_Nos(i) = Triangular_Nos(i-1) + i
      END DO
    END FUNCTION Triangular_Nos
    PROGRAM Blob
     IMPLICIT NONE
     INTERFACE
      FUNCTION Triangular_Nos(n)
       INTEGER, INTENT(IN)   :: n
       INTEGER, DIMENSION(n) :: Triangular_Nos
      END FUNCTION Triangular_Nos
     END INTERFACE
      PRINT*, Triangular_Nos(23)
      ! etc
    END
