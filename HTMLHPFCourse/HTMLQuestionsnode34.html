Here is a program to do this, 
    PROGRAM Draughts
     IMPLICIT NONE
     INTEGER :: i
     INTEGER, DIMENSION(8,8) :: Board = 0
     Board(::2,::2) = 1
     Board(2::2,2::2) = 1
     DO I = 1,8
      PRINT*, Board(i,:)
     END DO
    END PROGRAM Draughts
