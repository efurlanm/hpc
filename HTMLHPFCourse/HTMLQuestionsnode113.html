!HPF$ INDEPENDENT, NEW(alpha)
         DO j = i+1, n
            alpha = A(j,i) / swap(i)
            A(j, i+1:n) = A(j, i+1:n ) - alpha * SWAP(i+1:n)
            Y(j) = Y(j) - alpha * TEMP
            A(j,i) = alpha
         ENDDO
