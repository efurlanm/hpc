Solution

    PROGRAM Math_Majik
     IMPLICIT NONE
     INTEGER num, tnum
      outa: DO
       PRINT*, "Type in your number (0 terminates)"
       READ*, num
       IF (num .LE. 0) EXIT
       inna: DO
        tnum = num/2
        IF (2*tnum .EQ. num) THEN ! num even
         num = tnum
        ELSE ! Odd
         num = 3*num+1
        END IF
        PRINT*, num
        IF (num == 1) THEN
         PRINT*, "Sequence finishes nicely"
         EXIT
        ELSE IF (num == 13) THEN
         PRINT*, "Yoiks, Extreme bad luck encountered - Exit"
         EXIT outa
        END IF
       END DO inna
      END DO outa
    END PROGRAM Math_Majik
