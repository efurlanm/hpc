Program Example

    PROGRAM Main
     IMPLICIT NONE
     REAL x
     INTRINSIC FLOOR
     REAL, EXTERNAL :: TenTimes
      READ*, x
      PRINT*, FLOOR(x)
      PRINT*, Negative(x)
      PRINT*, TenTimes(x)
    CONTAINS
     REAL FUNCTION Negative(a)
      REAL, INTENT(IN) :: a
       Negative = -a
     END FUNCTION Negative
    END PROGRAM Main
    
    REAL FUNCTION TenTimes(a)
     IMPLICIT NONE
     REAL, INTENT(IN) :: a
      TenTimes = 10.0*a
    END FUNCTION TenTimes
