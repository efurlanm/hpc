    PROGRAM x_y
     IMPLICIT NONE
      REAL :: r, theta, pi, theta_in_radians
       pi = ATAN(1.0)*4.0
       PRINT*, "Type in r and Theta (in degrees)"
       READ*, r, theta
       theta_in_radians = (theta/180.0)*pi
       PRINT*, "( x,y ) is ( ",r*COS(theta_in_radians),&
              ",",r*SIN(theta_in_radians)," )"
    END PROGRAM x_y
