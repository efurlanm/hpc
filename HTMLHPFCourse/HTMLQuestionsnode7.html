Solution

 PROGRAM Area_Volume
  REAL :: radius, area, pi
  pi = ATAN(1.0)*4.0
  DO
   PRINT*, "Type in the radius, a negative value will terminate"
   READ*, radius
   IF (radius .LT. 0) EXIT
    area = pi*radius*radius
    Print*, "Area of circle with radius ",&
            radius, " is ", area
    Print*, "Volume of sphere with radius ",&
            radius, " is ", ((4.0/3.0)*radius)*area
  END DO
 END PROGRAM Area_Volume
